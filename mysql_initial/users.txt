CREATE DATABASE gms; 

CREATE TABLE Customer (
    Customer_id INT(10) AUTO_INCREMENT,
    Email varchar(255) NOT NULL,
    LastName varchar(255) NOT NULL,
    FirstName varchar(255) NOT NULL,
    Age int(4),
    Gender varchar(10),
    Telephone varchar(100),
    password varchar(255) NOT NULL,
    Details MEDIUMTEXT,
    Suspended INT(1) DEFAULT 0,
    Delected INT(1) DEFAULT 0,
    PRIMARY KEY (Customer_id)
);

CREATE TABLE Coach (
    Coach_id INT(10) AUTO_INCREMENT,
    Email varchar(255) NOT NULL,
    LastName varchar(255) NOT NULL,
    FirstName varchar(255) NOT NULL,
    City varchar(255) NOT NULL,
    Age int(4) NOT NULL,
    Gender varchar(10) NOT NULL,
    Telephone varchar(100) NOT NULL,
    password varchar(255) NOT NULL,
    Details MEDIUMTEXT,
    Suspended INT(1) DEFAULT 0,
    Delected INT(1) DEFAULT 0,
    PRIMARY KEY (Coach_id)
);

CREATE TABLE Admin (
    Admin_id INT(10) AUTO_INCREMENT,
    Email varchar(255) NOT NULL,
    LastName varchar(255) NOT NULL,
    FirstName varchar(255) NOT NULL,
    City varchar(255) NOT NULL,
    Age int(4) NOT NULL,
    Gender varchar(10) NOT NULL,
    Telephone varchar(100) NOT NULL,
    password varchar(255) NOT NULL,
    Details MEDIUMTEXT,
    Delected INT(1) DEFAULT 0,
    PRIMARY KEY (Admin_id)
);


CREATE TABLE Coach_Registration (
    Registration_id INT(10) AUTO_INCREMENT,
    Customer varchar(255) NOT NULL,
    Coach varchar(255) NOT NULL,
    Delected INT(1) DEFAULT 0,
    Details MEDIUMTEXT,
    PRIMARY KEY (Registration_id)
);
